Usage

Basic Example

import { ImportSDK } from './import-sdk.js';

const sdk = ImportSDK.init(document.getElementById('import-container'), {
  apiEndpoint: 'http://localhost:3000/api/import',
  chunkSize: 100,
  updateByTankNumber: false,
  fieldMapping: {
    'Tank ID': 'tankNumber',
    'Type': 'typeId'
  },
  transformers: {
    typeId: (value) => parseInt(value)
  },
  filters: {
    tankNumber: (value) => value && value.length > 0
  }
});
